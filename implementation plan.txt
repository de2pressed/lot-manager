# Raptile Studio — Ops Hub: Full Build Specification

## Project Context
You are building a production-ready, team-facing operations app for Raptile Studio, an Indian 
streetwear brand. The app merges two existing standalone browser apps (Lot Counter and Inventory 
Tracker) into one unified tool. The final result is deployed on Vercel (Vite + vanilla JS) with 
Supabase as the backend (Postgres, Auth, Realtime). No React, no Vue — vanilla JS, modular ES 
modules, clean DOM-driven architecture.

---

## Tech Stack

- **Frontend**: Vite, vanilla JS (ES modules), CSS custom properties
- **Backend**: Supabase (Postgres, Auth, Realtime, Storage)
- **Deployment**: Vercel (SPA + one serverless API function)
- **Charts**: Chart.js (CDN)
- **Export**: html2canvas (lot summary PNG), Papa Parse (sales CSV)
- **Shopify sync**: Client-side fetch to /products.json

-- Lot status auto-update trigger
create or replace function update_lot_status()
returns trigger as $$
declare
  total_qty integer;
  max_cap integer;
begin
  select sum(qty) into total_qty from public.lot_items where lot_id = NEW.lot_id;
  select max_items into max_cap from public.lots where id = NEW.lot_id;
  if total_qty >= max_cap then
    update public.lots set status = 'full' where id = NEW.lot_id and status = 'open';
  else
    update public.lots set status = 'open' where id = NEW.lot_id and status = 'full';
  end if;
  return NEW;
end;
$$ language plpgsql;

create trigger lot_items_capacity after insert or update or delete on public.lot_items
  for each row execute function update_lot_status();

-- Inventory status auto-update trigger
create or replace function update_inventory_status()
returns trigger as $$
begin
  if NEW.quantity <= 0 then
    NEW.status := 'sold_out';
  elsif NEW.quantity <= 3 then
    NEW.status := 'low_stock';
  else
    NEW.status := 'in_stock';
  end if;
  return NEW;
end;
$$ language plpgsql;

create trigger inventory_status_update before insert or update on public.inventory
  for each row execute function update_inventory_status();

-- Profile auto-creation on signup
create or replace function handle_new_user()
returns trigger as $$
begin
  insert into public.profiles (id, username, role)
  values (NEW.id, NEW.email, 'viewer');
  return NEW;
end;
$$ language plpgsql security definer;

create trigger on_auth_user_created after insert on auth.users
  for each row execute function handle_new_user();

## Row Level Security (RLS) Policies
```sql
-- Enable RLS on all tables
alter table public.profiles enable row level security;
alter table public.products enable row level security;
alter table public.lots enable row level security;
alter table public.lot_items enable row level security;
alter table public.inventory enable row level security;
alter table public.sales enable row level security;
alter table public.activity_log enable row level security;

-- Helper function to get current user role
create or replace function get_my_role()
returns text as $$
  select role from public.profiles where id = auth.uid();
$$ language sql security definer stable;

-- Profiles: users can read all, only admin can write
create policy "profiles_read_all" on public.profiles
  for select using (auth.role() = 'authenticated');
create policy "profiles_admin_write" on public.profiles
  for all using (get_my_role() = 'admin');

-- Products: everyone reads, manager+ writes
create policy "products_read" on public.products
  for select using (auth.role() = 'authenticated');
create policy "products_write" on public.products
  for all using (get_my_role() in ('admin','manager'));

-- Lots: everyone reads, manager+ and ops can write
create policy "lots_read" on public.lots
  for select using (auth.role() = 'authenticated');
create policy "lots_write" on public.lots
  for all using (get_my_role() in ('admin','manager','ops'));

-- Lot items: everyone reads, manager+ and ops write
create policy "lot_items_read" on public.lot_items
  for select using (auth.role() = 'authenticated');
create policy "lot_items_write" on public.lot_items
  for all using (get_my_role() in ('admin','manager','ops'));

-- Inventory: everyone reads, manager+ writes
create policy "inventory_read" on public.inventory
  for select using (auth.role() = 'authenticated');
create policy "inventory_write" on public.inventory
  for all using (get_my_role() in ('admin','manager'));

-- Sales: everyone reads, manager+ writes
create policy "sales_read" on public.sales
  for select using (auth.role() = 'authenticated');
create policy "sales_write" on public.sales
  for all using (get_my_role() in ('admin','manager'));

-- Activity log: everyone reads, all authenticated can insert
create policy "log_read" on public.activity_log
  for select using (auth.role() = 'authenticated');
create policy "log_insert" on public.activity_log
  for insert with check (auth.role() = 'authenticated');
```

## Vercel Serverless Function: User Creation

File: `api/admin/create-user.js`
```js
import { createClient } from '@supabase/supabase-js';

const supabaseAdmin = createClient(
  process.env.SUPABASE_URL,
  process.env.SUPABASE_SERVICE_KEY  // never expose this to the client
);

export default async function handler(req, res) {
  if (req.method !== 'POST') return res.status(405).end();

  // Verify the caller is an authenticated admin
  const authHeader = req.headers.authorization;
  if (!authHeader) return res.status(401).json({ error: 'Unauthorized' });

  const token = authHeader.replace('Bearer ', '');
  const { data: { user }, error: authError } = await supabaseAdmin.auth.getUser(token);
  if (authError || !user) return res.status(401).json({ error: 'Invalid token' });

  const { data: profile } = await supabaseAdmin
    .from('profiles')
    .select('role')
    .eq('id', user.id)
    .single();

  if (!profile || profile.role !== 'admin') {
    return res.status(403).json({ error: 'Admin only' });
  }

  const { email, password, username, role } = req.body;
  if (!email || !password || !username || !role) {
    return res.status(400).json({ error: 'Missing fields' });
  }

  // Create auth user
  const { data: newUser, error: createError } = await supabaseAdmin.auth.admin.createUser({
    email,
    password,
    email_confirm: true
  });

  if (createError) return res.status(500).json({ error: createError.message });

  // Update their profile with username and role (trigger creates the profile row)
  const { error: profileError } = await supabaseAdmin
    .from('profiles')
    .update({ username, role, created_by: user.id })
    .eq('id', newUser.user.id);

  if (profileError) return res.status(500).json({ error: profileError.message });

  return res.status(200).json({ success: true, userId: newUser.user.id });
}
```

Vercel env vars needed (set in Vercel dashboard, not in .env):
- `SUPABASE_URL` 
- `SUPABASE_SERVICE_KEY` (service role key — never in client)

## The Core Feature: Lot → Inventory Push

This is the most important flow. Implement it as a transaction in `lots.service.js`.
```js
// src/services/lots.service.js

export async function pushLotToInventory(lotId, userId) {
  // 1. Fetch lot + all lot_items
  const { data: lot } = await supabase
    .from('lots')
    .select('*, lot_items(*, products(*))')
    .eq('id', lotId)
    .single();

  if (!lot || lot.status === 'pushed') throw new Error('Lot already pushed or not found');

  // 2. Build inventory upsert rows — group lot_items by product+variant to merge qty
  const inventoryMap = new Map();
  for (const item of lot.lot_items) {
    const key = `${item.product_id}_${item.variant_title}`;
    if (inventoryMap.has(key)) {
      inventoryMap.get(key).quantity += item.qty;
    } else {
      inventoryMap.set(key, {
        lot_id: lotId,
        lot_item_id: item.id,
        product_id: item.product_id,
        shopify_product_id: item.shopify_product_id,
        product_title: item.product_title,
        variant_title: item.variant_title,
        color: item.color,
        size: item.size,
        sku: item.sku,
        buy_price: item.buy_price ?? 0,
        quantity: item.qty,
        created_by: userId
      });
    }
  }

  const inventoryRows = [...inventoryMap.values()];

  // 3. Upsert into inventory — if same lot+product+variant already exists, add qty
  const { error: invError } = await supabase
    .from('inventory')
    .upsert(inventoryRows, {
      onConflict: 'lot_id,product_id,variant_title',
      ignoreDuplicates: false
    });

  if (invError) throw invError;

  // 4. Mark lot as pushed
  await supabase
    .from('lots')
    .update({ status: 'pushed', pushed_at: new Date().toISOString(), pushed_by: userId })
    .eq('id', lotId);

  // 5. Log it
  await logActivity({
    userId,
    type: 'lot_pushed',
    description: `Lot "${lot.name}" pushed to inventory (${lot.lot_items.length} variants)`,
    refId: lotId,
    refType: 'lot'
  });

  return { pushed: inventoryRows.length };
}
```

The UI for this: on the Lots page, each lot card has a "Push to Inventory" button that is:
- Disabled and labelled "Already Pushed" if `status === 'pushed'`
- Active with a confirm modal for open/full lots
- Only visible to admin and manager roles

## Realtime Subscriptions

Set up in layout.js after mount, tear down on sign-out.
```js
const channels = [];

export function subscribeToRealtime() {
  channels.push(
    supabase.channel('inventory-changes')
      .on('postgres_changes', { event: '*', schema: 'public', table: 'inventory' },
        payload => state.emit('inventory:changed', payload))
      .subscribe(),

    supabase.channel('sales-changes')
      .on('postgres_changes', { event: '*', schema: 'public', table: 'sales' },
        payload => state.emit('sales:changed', payload))
      .subscribe(),

    supabase.channel('lots-changes')
      .on('postgres_changes', { event: '*', schema: 'public', table: 'lots' },
        payload => state.emit('lots:changed', payload))
      .subscribe()
  );
}

export function unsubscribeAll() {
  channels.forEach(c => c.unsubscribe());
  channels.length = 0;
}
```

Views listen to `state.on('inventory:changed', ...)` and re-render their tables reactively.

## State Management

Use a minimal pub/sub event bus. No Redux, no Zustand.
```js
// src/state.js
const listeners = new Map();

export const state = {
  // In-memory cache (populated from Supabase on mount, kept fresh via realtime)
  lots: [],
  inventory: [],
  sales: [],
  products: [],
  log: [],
  currentUser: null,
  currentRole: null,

  emit(event, data) {
    (listeners.get(event) || []).forEach(fn => fn(data));
  },
  on(event, fn) {
    if (!listeners.has(event)) listeners.set(event, []);
    listeners.get(event).push(fn);
    return () => {
      const fns = listeners.get(event) || [];
      listeners.set(event, fns.filter(f => f !== fn));
    };
  }
};
```

## localStorage Migration

File: `src/migrations/localStorage.js`
```js
export async function checkLocalStorageMigration(userId) {
  const oldLotData = localStorage.getItem('lotCounter');
  const oldInvData = localStorage.getItem('raptile_inventory_v1');
  const alreadyMigrated = localStorage.getItem('raptile_migrated_v2');

  if (alreadyMigrated || (!oldLotData && !oldInvData)) return;

  // Show a migration offer modal
  const confirmed = await showMigrationModal(
    Boolean(oldLotData),
    Boolean(oldInvData)
  );

  if (!confirmed) {
    localStorage.setItem('raptile_migrated_v2', 'skipped');
    return;
  }

  if (oldLotData) await migrateLots(JSON.parse(oldLotData), userId);
  if (oldInvData) await migrateInventory(JSON.parse(oldInvData), userId);

  localStorage.setItem('raptile_migrated_v2', 'done');
}

async function migrateLots(oldState, userId) {
  // Map old lot structure to new schema
  // old: { lots: [{ id, name, maxItems, items, createdAt }] }
  for (const lot of (oldState.lots || [])) {
    const { data: newLot } = await supabase
      .from('lots')
      .insert({ name: lot.name, max_items: lot.maxItems, created_by: userId })
      .select().single();

    const items = lot.items.map(item => ({
      lot_id: newLot.id,
      product_title: item.productName,
      variant_title: [item.color, item.size].filter(Boolean).join(' / '),
      color: item.color,
      size: item.size,
      qty: item.qty
    }));

    if (items.length) {
      await supabase.from('lot_items').insert(items);
    }
  }
}

async function migrateInventory(oldState, userId) {
  // Map old inventory structure to new schema
  for (const item of (oldState.inventory || [])) {
    await supabase.from('inventory').insert({
      product_title: item.name,
      variant_title: item.variant || '',
      sku: item.sku,
      buy_price: item.buyPrice || 0,
      quantity: item.quantity || 0,
      notes: item.notes,
      created_by: userId
    });
  }
  for (const sale of (oldState.sales || [])) {
    await supabase.from('sales').insert({
      product_title: sale.name,
      variant_title: sale.variant || '',
      buy_price: sale.buyPrice,
      sale_price: sale.salePrice,
      qty_sold: sale.qtySold,
      platform: sale.platform,
      date_sold: sale.dateSold
    });
  }
}
```

## Shopify Product Sync

File: `src/services/shopify.js`
```js
export async function syncShopifyProducts() {
  const storeUrl = 'https://raptilestudio.myshopify.com';
  let allProducts = [];
  let page = 1;
  let hasMore = true;

  while (hasMore) {
    try {
      const res = await fetch(`${storeUrl}/products.json?limit=250&page=${page}`);
      const { products } = await res.json();
      if (!products.length) { hasMore = false; break; }
      allProducts = allProducts.concat(products);
      page++;
    } catch {
      hasMore = false;
    }
  }

  // Normalize and upsert to Supabase products table
  const rows = allProducts.map(p => ({
    shopify_product_id: p.id,
    title: p.title,
    image_url: p.images?.[0]?.src ?? null,
    variants: p.variants.map(v => ({
      id: v.id,
      title: v.title,
      sku: v.sku,
      price: parseFloat(v.price),
      color: extractOption(p, v, ['color','colour']),
      size: extractOption(p, v, ['size'])
    })),
    synced_at: new Date().toISOString()
  }));

  const { error } = await supabase
    .from('products')
    .upsert(rows, { onConflict: 'shopify_product_id' });

  if (error) throw error;
  return rows.length;
}

function extractOption(product, variant, names) {
  for (const name of names) {
    const opt = product.options.find(o => o.name.toLowerCase() === name);
    if (!opt) continue;
    const pos = opt.position;
    return variant[`option${pos}`] ?? null;
  }
  return null;
}
```

## Shared UI Conventions

**Sidebar navigation** — these views, in order:
1. Dashboard
2. Lots (includes "Push to Inventory" action)
3. Products (Shopify catalog + manual)
4. Inventory
5. Sales (To Sell tab + Sold tab)
6. History
7. Admin (visible to admin role only)
8. Settings

**Role-gated rendering**: Every view that requires a specific role should call `guardRole(['admin','manager'])` at the top of its render function, which redirects to dashboard with a toast if unauthorized.

**Toast types**: success, error, warning, info. Auto-dismiss after 4 seconds. Max 3 visible at once.

**All modals** must:
- Use a single shared `modal.js` open/close utility (not separate modal HTML per view)
- Close on Escape key and overlay click
- Trap focus within the modal while open

**Keyboard shortcuts** (global):
- `N` — open new lot modal (if on Lots view)
- `N` — open product picker (if on Inventory view)
- `Esc` — close active modal
- `S` — trigger Shopify sync (Settings view only)

## Dashboard Analytics

Compute from Supabase data on mount (not on every render):

- Total revenue (sum of `sales.sale_price * qty_sold`)
- Total cost (sum of `sales.buy_price * qty_sold`)
- Total profit (sum of `sales.profit * qty_sold`)
- Profit margin (profit / revenue × 100)
- Units in stock (sum of `inventory.quantity` where status != 'sold_out')
- Units sold (sum of `sales.qty_sold`)
- Active lots (count of lots where status != 'pushed')
- Low stock alerts (inventory items where status = 'low_stock')

Charts (Chart.js):
1. Monthly revenue — last 6 months bar chart
2. Cumulative profit — line chart
3. Top 5 products by units sold — horizontal bar chart
4. Platform breakdown — doughnut chart (Direct, Instagram DM, WhatsApp, etc.)

Only visible when `currentRole === 'admin'`.

Features:
- Table of all users (from `profiles`) with their role and creation date
- "Create user" button → modal with email, username, password, role selector
- Role can be changed inline via a dropdown (calls update on `profiles` table)
- Delete user button (calls Supabase admin delete via the same serverless function with a `DELETE` endpoint)
- Show which user last pushed each lot

The user creation form sends:
```js
const session = await supabase.auth.getSession();
fetch('/api/admin/create-user', {
  method: 'POST',
  headers: {
    'Content-Type': 'application/json',
    'Authorization': `Bearer ${session.data.session.access_token}`
  },
  body: JSON.stringify({ email, password, username, role })
});
```

## Export Features

**Sales CSV** (via Papa Parse):

- Queries all tables and exports as a single JSON file

## Critical Constraints

- The service role key must NEVER appear in client-side code or VITE_ env vars
- RLS policies are the security layer — frontend role checks are UX only
- Every write operation must insert a row into `activity_log`
- The lot → inventory push is irreversible from the UI (admin can manually edit the DB)
- Realtime subscriptions should update the UI incrementally, not re-fetch all data
- The app must work gracefully if Shopify sync fails (use cached products from Supabase)
- All modals must be accessible (focus trap, Escape to close, ARIA roles)
